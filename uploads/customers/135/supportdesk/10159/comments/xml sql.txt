SELECT * FROM 
    			(SELECT MAX  (OD.SHIPPINGMARKCODE) SHIPPINGMARKCODE, O.CONSIGNEEKEY, O.C_CONTACT1, O.C_CONTACT2, O.C_ADDRESS1, O.C_ADDRESS2,  
    			 O.C_ADDRESS3, O.C_ZIP, O.C_CITY, O.C_COUNTRY, MAX(OD.CUSTOMERORDERNO) CUSTOMERORDERNO,  D.CASENO, D.MAINACCNO,  
    			 D.CCUBE, D.TWEIGHT,D.CWEIGHT, D.DRUDF1 LENGTH, D.DRUDF2 WIDTH, D.DRUDF3 HEIGHT,  
    			 ROW_NUMBER() OVER (ORDER BY  D.CASENO) as row
    			 FROM ORDERS O, ORDERDETAIL OD, DROPID D, DROPIDDETAIL DD, PICKDETAIL PD 
    			 WHERE D.MAINACCNO = OD.SUSR1 and DD.dropid = D.dropid and DD.CHILDID = PD.CASEID and PD.orderkey = OD.orderkey and PD.orderlinenumber = OD.orderlinenumber 
    			 AND OD.ORDERKEY = O.ORDERKEY 
    			 AND D.CASENO in (SELECT CASEID from LOADORDERDETAIL , LOADHDR , LOADSTOP WHERE LOADSTOP.LOADSTOPID = LOADORDERDETAIL.LOADSTOPID and LOADSTOP.LOADID = LOADHDR.LOADID and LOADHDR.LOADID = '0000004423') 
    			 AND D.ISCLOSED = 'Y' 
    			 GROUP BY O.CONSIGNEEKEY, O.C_CONTACT1, O.C_CONTACT2,  O.C_ADDRESS1, O.C_ADDRESS2, O.C_ADDRESS3, 
    			 O.C_ZIP, O.C_CITY, O.C_COUNTRY, D.CASENO,  D.MAINACCNO, D.CCUBE, D.TWEIGHT,D.CWEIGHT, D.DRUDF1 , D.DRUDF2 , D.DRUDF3
    			) A
